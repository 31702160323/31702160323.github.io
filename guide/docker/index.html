# Docker 的常用笔记

# 帮助命令

docker  version		# 显示`docker`的版本号

docker info			# 显示`docker`的系统信息

docker --help		# 帮助命令

# 镜像命令

docker images # 查看所有本地的主机上的镜像
REPOSITORY          TAG                 IMAGE ID            CREATED             SIZE
hello-world         latest              bf756fb1ae65        11 months ago       13.3kB

# 解释
REPOSITORY	# 镜像的仓库源
TAG			# 镜像的标签
IMAGE ID	# 镜像的id
CREATED		# 镜像的创建时间
SIZE		# 镜像的大小

# 可选项
-a --a 		# 列出所有的镜像
-q --quiet	# 只显示镜像的id
# 镜像搜索命令
docker search XXX

docker search XXX --filter=STARS=3000 # 搜索STARS大于3000的
# 镜像下载
# docker pull 镜像名:版本号
# 删除镜像
docker rmi -f 镜像id或名字
# 创建镜像
docker build -t 镜像名:tag Dockerfile路径

# 容器命令

# 新建容器并启动
docker run [可选参数] [镜像名]

##可选参数
--name="Name"	# 容器的名字
-d				# 后台运行
-it				# 使用交互方式运行，进入容器查看内容
-p				# 容器随机端口
-p				# 指定容器的端口 -p 8080:8080
-v				# 映射文件路径 宿主路径:容器路径  不支持相对路径
	-p id:主机端口:容器端口
	-p 主机端口:容器端口  # 常用
	-p 容器端口
	容器端口
# 列出所有运行的容器
docker ps [可选参数]
			# 列出当前运行的容器
	-a		# 列出当前运行的容器 + 带出历史运行的容器
	-n=?	# 显示最近创建的容器
	-q		# 只显示容的编号
# 退出容器
exit			# 从容器退回主机
Ctr + P + Q		# 容器不停止退出
# 删除容器
docker rm 容器 id						# 删除指定的容器
docker rm -f $(docker ps -aq)		 # 删除所有的容器
# 启动和停止容器的操作
docker start 容器id			# 启动容器
docker restart 容器id			# 重庆容器
docker stop 容器id			# 停止容器
docker kill 容器id			# 强制停止容器

# 常用其他命令

# 后台启动容器
# 问题 “docker run -d 镜像名” 运行容器后， “docker ps” 查看容器，发现容器停止了

# 常见的坑，docker 容器使用后台运行，就必须要有一个前台进程，docker 发现没有应用了，就会自动停止
# nginx,容器启动后， 发现自己没有提供服务，就会立即停止，就是因为没有程序了
# 日志命令
docker logs -f -t --tail 条数 容器id
# 查看容器进程信息
docker top 容器
# 查看镜像的元信息
docker inspect 容器id
# 进入当前运行容器
# 我们通常容器都是使用后台运行的，使用进入容器，修改一些配置

## 方式1
docker exec -it 容器id bashShell  ## /bin/bash

## 方式2
docker attach 容器id

## 区别
### docker exect 进入容器后开启一个新的终端，可以在里面操作 （常用）
### docker attach 进入容器正在执行的终端，不开启新的终端
# 从容器拷贝文件到主机
docker cp 容器id:容器内路径 目的的主机路径
