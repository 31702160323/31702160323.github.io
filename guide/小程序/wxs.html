# wxs

<template>
	<view>
		<view class="tabs" :change:prop="test.tab" :prop="current">
			<template v-for="(item, index) in tabList">
				<view class="tabs-item" @click="tab(index)">{{ item }}</view>
			</template>
			<view class="tabs-link"></view>
		</view>
		<view>
			{{ current }}
		</view>
	</view>
</template>
<script module="test" lang="wxs">
	module.exports = {
		tab: function(newValue, oldValue, ownerInstance, instance) {
			var item = ownerInstance.selectAllComponents('.tabs-item')[newValue].getBoundingClientRect();
			var instance = ownerInstance.selectComponent('.tabs-link');
			var rect = instance.getBoundingClientRect();
			instance.setStyle({
				width: item.width * 0.6 + 'px'
				left: item.left + ((item.width - (rect.width === 0 ? item.width * 0.8 : rect.width)) / 2) + 'px'
			})
			return false
		}
	}
</script>

<script>
	export default {
		data() {
			return {
				current: 0,
				tabList: ['首页', '发现', '我的']
			}
		},
		methods: {
			tab(index) {
				this.current = index
			}
		}
	}
</script>
